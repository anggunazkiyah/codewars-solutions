// Codewars - Additive Inverse (8kyu)

function opposite(number) {
  //your code here
  return (-number);
}

console.log(opposite(7)); // Output: -7
console.log(opposite(3 * -1)); // Output: 3 * -1 = -3, so opposite(-3) = 3
console.log(opposite(0.100 * 8)); // Output: 0.1 * 8 = 0.8, so opposite(0.8) = -0.8
console.log(opposite(-4)); // Output: 4
console.log(opposite(-4 * -4)) // Output: -4 * -4 = 16, so opposite(16) = -16
